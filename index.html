def train_naive_bayes(freqs, train_x, train_y):

    loglikelihood = {}
    logprior = 0

    # Vset
    vocab = set([pair[0] for pair in freqs.keys()])
    V = len(vocab)

    # N_actN_ske
    N_act = N_ske = 0
    for pair in freqs.keys():
        # 1activist(if the label is activist == 1)
        if pair[1] > 0:
            # Increment the number of positive words by the count for this (word, label) pair
            N_act += freqs[pair]
        # skeptic
        else:
            # increment the number of negative words by the count for this (word,label) pair
            N_ske += freqs[pair]

    # 
    D = len(train_y)

    # activist
    D_act = len([x for x in train_y if x > 0])

    # skeptic
    D_ske = len([x for x in train_y if x <= 0])

    # logprior
    logprior = np.log(D_act) - np.log(D_ske)

    # activistskeptic
    for word in vocab:
        freq_act = lookup(freqs, word, 1)
        freq_ske = lookup(freqs, word, 0)

        # activistskeptic
        p_w_act = (freq_act + 1) / (N_act + V)
        p_w_ske = (freq_ske + 1) / (N_ske + V)

        #  log likelihood of the word
        loglikelihood[word] = np.log(p_w_act/p_w_ske)

    return logprior, loglikelihood
